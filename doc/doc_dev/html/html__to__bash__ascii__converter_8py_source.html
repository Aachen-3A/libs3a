00324 def rgb2short(rgb):
00325     """ Find the closest xterm-256 approximation to the given RGB value.
00326     @param rgb: Hex code representing an RGB value, eg, 'abcdef'
00327     @returns: String between 0 and 255, compatible with xterm.
00328     >>> rgb2short('123456')
00329     ('23', '005f5f')
00330     >>> rgb2short('ffffff')
00331     ('231', 'ffffff')
00332     >>> rgb2short('0DADD6') # vimeo logo
00333     ('38', '00afd7')
00334     """
00335     rgb = _strip_hash(rgb)
00336     incs = (0x00, 0x5f, 0x87, 0xaf, 0xd7, 0xff)
00337     # Break 6-char RGB code into 3 integer vals.
00338     parts = [ int(h, 16) for h in re.split(r'(..)(..)(..)', rgb)[1:4] ]
00339     res = []
00340     for part in parts:
00341         i = 0
00342         while i < len(incs)-1:
00343             s, b = incs[i], incs[i+1]  # smaller, bigger
00344             if s <= part <= b:
00345                 s1 = abs(s - part)
00346                 b1 = abs(b - part)
00347                 if s1 < b1: closest = s
00348                 else: closest = b
00349                 res.append(closest)
00350                 break
00351             i += 1
00352     #print '***', res
00353     res = ''.join([ ('%02.x' % i) for i in res ])
00354     equiv = RGB2SHORT_DICT[ res ]
00355     #print '***', res, equiv
00356     return equiv, res
